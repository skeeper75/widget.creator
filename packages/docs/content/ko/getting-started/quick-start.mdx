### 방법 2: ESM Import (React/Vue 프로젝트)

#### 1. npm 패키지 설치

```bash
npm install @huni/widget-core
# 또는
pnpm add @huni/widget-core
```

#### 2. React 컴포넌트에서 사용

```tsx
import React, { useEffect, useRef, useState } from 'react';
import { HuniWidget } from '@huni/widget-core';

function PrintQuoteWidget() {
    const containerRef = useRef<HTMLDivElement>(null);
    const [quote, setQuote] = useState<number>(0);

    useEffect(() => {
        if (!containerRef.current) return;

        const widget = new HuniWidget({
            containerId: 'huni-widget-react',
            apiBaseUrl: 'https://api.huni.builder',
            locale: 'ko-KR',
            productId: 'postcard-001',
            onQuoteChange: (newQuote) => {
                setQuote(newQuote.totalPrice);
            }
        });

        widget.mount();

        return () => {
            widget.unmount();
        };
    }, []);

    return (
        <div style={{ maxWidth: '600px', margin: '50px auto' }}>
            <h1>인쇄 견적 위젯</h1>
            <div style={{ fontSize: '24px', fontWeight: 'bold', textAlign: 'center', color: '#007bff' }}>
                {quote.toLocaleString()}원
            </div>
            <div id="huni-widget-react" ref={containerRef} />
        </div>
    );
}

export default PrintQuoteWidget;
```

#### 3. Vue 컴포넌트에서 사용

```vue
<template>
  <div style="max-width: 600px; margin: 50px auto;">
    <h1>인쇄 견적 위젯</h1>
    <div style="font-size: 24px; font-weight: bold; text-align: center; color: #007bff;">
      {{ quote.toLocaleString() }}원
    </div>
    <div id="huni-widget-vue" ref="widgetContainer"></div>
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue';
import { HuniWidget } from '@huni/widget-core';

const widgetContainer = ref<HTMLDivElement>();
const quote = ref<number>(0);
let widget: HuniWidget | null = null;

onMounted(() => {
  if (!widgetContainer.value) return;

  widget = new HuniWidget({
    containerId: 'huni-widget-vue',
    apiBaseUrl: 'https://api.huni.builder',
    locale: 'ko-KR',
    productId: 'postcard-001',
    onQuoteChange: (newQuote) => {
      quote.value = newQuote.totalPrice;
    }
  });

  widget.mount();
});

onUnmounted(() => {
  widget?.unmount();
});
</script>
```

### 방법 3: Shopby 연동

Shopby 쇼핑몰에 위젯을 연동하는 방법입니다.

#### 1. HuniWidgetMount.jsx 설정

```jsx
// ref/shopby/aurora-skin(dev)/HuniWidget/HuniWidgetMount.jsx
import React, { useEffect, useRef } from 'react';

const HuniWidgetMount = ({ mesCode, shopbyToken }) => {
    const containerRef = useRef(null);

    useEffect(() => {
        if (!containerRef.current) return;

        // 위젯 IIFE 스크립트 로드
        const script = document.createElement('script');
        script.src = 'https://huni.builder/widget/huni-widget.iife.js';
        script.onload = () => {
            // 위젯 마운트
            huniWidget.mount({
                target: containerRef.current,
                apiBaseUrl: 'https://api.huni.builder',
                locale: 'ko-KR',
                shopbyToken: shopbyToken,
                mesCode: mesCode,
                onQuoteChange: (quote) => {
                    // CustomEvent로 Shopby에 견적 변경 전달
                    window.dispatchEvent(new CustomEvent('huni-quote-change', {
                        detail: { quote }
                    }));
                }
            });
        };
        document.head.appendChild(script);

        return () => {
            // Cleanup
            if (window.huniWidget) {
                window.huniWidget.unmount();
            }
        };
    }, [mesCode, shopbyToken]);

    return <div ref={containerRef} />;
};

export default HuniWidgetMount;
```

#### 2. Purchase.jsx에서 위젯 렌더링

```jsx
// Shopby Purchase.jsx에서 위젯 사용
import HuniWidgetMount from './HuniWidget/HuniWidgetMount';

function Purchase({ product, shopbyToken }) {
    const mesCode = product.customOptions.huniMesCode;

    return (
        <div>
            {/* 기존 상품 정보 */}
            <h1>{product.name}</h1>
            <p>{product.description}</p>

            {/* huni.builder 위젯 */}
            <HuniWidgetMount
                mesCode={mesCode}
                shopbyToken={shopbyToken}
            />
        </div>
    );
}
```

#### 3. CustomEvent 프로토콜로 통신

```jsx
// 위젯 -> Shopby 통신
useEffect(() => {
    const handleQuoteChange = (event) => {
        const { quote } = event.detail;
        // Shopby 장바구니에 가격 업데이트
        updateCartPrice(quote.totalPrice);
    };

    window.addEventListener('huni-quote-change', handleQuoteChange);

    return () => {
        window.removeEventListener('huni-quote-change', handleQuoteChange);
    };
}, []);
```
